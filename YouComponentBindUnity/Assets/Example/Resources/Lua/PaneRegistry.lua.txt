-- Global panel registry for lua-driven UI.
-- Load this file once, then each panel can register itself by name.

UIRegistry = UIRegistry or {}
UIPanelName = {}

local panelMap = {}

function UIRegistry.Register(panelName, panel)
    if type(panelName) ~= "string" or panelName == "" then
        error("[UIRegistry] Register failed: invalid panelName")
    end
    if panel == nil then
        error("[UIRegistry] Register failed: panel is nil")
    end

    panelMap[panelName] = panel
    return panel
end

function UIRegistry.Get(panelName)
    return panelMap[panelName]
end

function UIRegistry.Unregister(panelName, panel)
    local current = panelMap[panelName]
    if current == nil then
        return false
    end

    if panel ~= nil and current ~= panel then
        return false
    end

    panelMap[panelName] = nil
    return true
end

function UIRegistry.Has(panelName)
    return panelMap[panelName] ~= nil
end

function UIRegistry.Clear()
    for key in pairs(panelMap) do
        panelMap[key] = nil
    end
end

function UIRegistry.GetMap()
    return panelMap
end

function CallLuaPanelFunction(luaWindowName, functionName, luaObject)
    if type(luaWindowName) ~= "string" or luaWindowName == "" then
        error("[CallLuaPanelFunction] invalid luaWindowName")
    end
    if type(functionName) ~= "string" or functionName == "" then
        error("[CallLuaPanelFunction] invalid functionName")
    end
    if type(UIPanelName) ~= "table" then
        error("[CallLuaPanelFunction] UIPanelName is not ready")
    end

    local panelName = UIPanelName[luaWindowName]
    if type(panelName) ~= "string" or panelName == "" then
        error("[CallLuaPanelFunction] panel name not found: " .. luaWindowName)
    end

    local panel = UIRegistry.Get(panelName)
    if panel == nil then
        error("[CallLuaPanelFunction] panel not found: " .. panelName)
    end

    local targetFunc = panel[functionName]
    if targetFunc == nil or type(targetFunc) ~= "function" then
        error("[CallLuaPanelFunction] function not found: " .. luaWindowName .. "." .. functionName)
    end

    if luaObject == nil then
        return targetFunc(panel)
    end

    return targetFunc(panel, luaObject)
end

function CleanupLuaPanelCallbacks()
    for _, panel in pairs(panelMap) do
        if panel ~= nil and type(panel.UnregisterEvent) == "function" then
            local ok, err = pcall(function()
                panel:UnregisterEvent()
            end)
            if not ok then
                print("[CleanupLuaPanelCallbacks] UnregisterEvent failed:", err)
            end
        end
    end

    UIRegistry.Clear()
end

return UIRegistry
