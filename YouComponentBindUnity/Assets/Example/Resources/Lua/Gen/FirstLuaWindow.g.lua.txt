function FirstLuaWindow_OnEnable()
    local panel = GetFirstLuaWindowPanel()
    if panel == nil then
        return
    end

    local view = panel.view
    if view == nil then
        print("[FirstLuaWindow] View is nil.")
        return
    end

    -- 防止重复注册
    view.Button_Button.onClick:RemoveListener(FirstLuaWindow_OnButton_ButtonClick)
    view.Toggle_Item.onValueChanged:RemoveListener(FirstLuaWindow_OnToggle_ItemValueChanged)
    view.Toggle_Toggle.onValueChanged:RemoveListener(FirstLuaWindow_OnToggle_ToggleValueChanged)

    view.Button_Button.onClick:AddListener(FirstLuaWindow_OnButton_ButtonClick)
    view.Toggle_Item.onValueChanged:AddListener(FirstLuaWindow_OnToggle_ItemValueChanged)
    view.Toggle_Toggle.onValueChanged:AddListener(FirstLuaWindow_OnToggle_ToggleValueChanged)

    if panel.Enable ~= nil then
        panel:Enable()
    end
end

function FirstLuaWindow_OnDisable()
    local panel = GetFirstLuaWindowPanel()

        return

    view.Button_Button.onClick:RemoveListener(FirstLuaWindow_OnButton_ButtonClick)
    view.Toggle_Item.onValueChanged:RemoveListener(FirstLuaWindow_OnToggle_ItemValueChanged)
    view.Toggle_Toggle.onValueChanged:RemoveListener(FirstLuaWindow_OnToggle_ToggleValueChanged)

    if panel.Disable ~= nil then
